28 void Cluster::modificar(const string &id_cpu, BinTree<string> &c, const BinTree<string> &c2, int &x) {
29  if (not c.empty()) {
30  if (c.value() == id_cpu) {
31  if (c.left().empty() and c.right().empty()) { //Si el CPU no contiene CPUs auxiliares, el nodose sutituye por el arbol c2.
32  x = 0;
33  c = c2;
34  }
35  else x = 3; //Contiene CPUs auxiliares.
36  }
37  else {
38  /*Como en la clase BinTree c.left() y c.right() son constantes y no se pueden modificar, creo dos arboles e y d,
39  tales que son identicos a c.left() y c.right() pero no son constantes, por lo cual puedo modificarlos.
40  */
41  BinTree<string> e, d;
42  e = c.left();
43  d = c.right();
44  modificar (id_cpu, e, c2, x);
45  if (x == 0) c = BinTree<string> (c.value(), e, d); //Si hemos logrado sustituir el nodo por c2, e(rama modificada) pasa a ser de c.
46  else if(x == 1) { //Miramos rama derecha d.
47  modificar (id_cpu, d, c2, x);
48  c = BinTree<string> (c.value(), e, d);
49  }
50  }
51  }
52  else x = 1; //No ha encontrado el CPU que coincida con el id_cpu.

109 int Cluster::miramos (const map<string, CPU> &cList, Parametros &aux, const Proceso &pro, const BinTree<string> &c, int p) {
110  if (not c.empty()) {
111  map<string, CPU>::const_iterator it = cList.find (c.value());
112  if (not (*it).second.existe(pro.consultar_id())) { //Si existe un proceso con el mismo id no entramos.
113  int x = (*it).second.espacio_para_proceso(pro.consultar_space()); //Devuelve el hueco mas ajustado que tenga el CPU para el proceso.
114  if (x >= pro.consultar_space()) {
115  int m = (*it).second.consultar_memoria_libre();
116  if (aux.hueco == -1) { //Primera iteración.
117  aux.id = (*it).first;
118  aux.hueco = x;
119  aux.mem = m;
120  aux.p = p;
121  }
122  else if (x < aux.hueco) {
123  aux.id = (*it).first;
124  aux.hueco = x;
125  aux.mem = m;
126  aux.p = p;
127  }
128  else if (x == aux.hueco) { //Empate, miramos mayor memoria libre.
129  if (m < aux.mem) {
130  aux.id = (*it).first;
131  aux.hueco = x;
132  aux.mem = m;
133  aux.p = p;
134  }
135  else if (m == aux.mem) { //Empate, miramos menor profundidad.
136  if (p < aux.p) {
137  aux.id = (*it).first;
138  aux.hueco = x;
139  aux.mem = m;
140  aux.p = p;
141  }
142 
143  }
144  }
145  }
146  }
147 
148  miramos (cList, aux, pro, c.left(), p+1);
149  miramos (cList, aux, pro, c.right(), p+1);
150  if(aux.hueco == -1) return -1; //Si no se ha encontrado ningún CPU que cumpla las características, devuelve -1.
151  else return 0;
152  }
153  else return -1;
154 }
